CREATE DATABASE MinhasFinancas;

USE MinhasFinancas;

CREATE TABLE infoFinancas (
    id INT PRIMARY KEY AUTO_INCREMENT,
    nome VARCHAR(255),
    salario FLOAT,
    mes VARCHAR(100)
);

CREATE TABLE gastos (
    id INT PRIMARY KEY AUTO_INCREMENT,
    tipoGasto VARCHAR(255),
    valorDoGasto FLOAT,
    idPessoa INT,
    FOREIGN KEY (idPessoa) REFERENCES infoFinancas(id)
);
